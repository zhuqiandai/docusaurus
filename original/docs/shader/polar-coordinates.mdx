## Map original point to center

```c
uniform sampler2D tex;

vec2 st = uv
uv *= 2.;
uv -= 1.;

float radius = length(uv);
float theta = atan(uv.y, uv.x);

/**
 *  pass polar coordinates to texture
 *  1. x = cos(theta) * radius
 *  2. y = sin(theta) * radius
 *
 *  这里再次转换成用 x,y 笛卡尔坐标系来表示
 *  但是可以控制的是，在转换过程中的 radius 和 theta 
 */

vec2 polar = vec2(radius * cos(theta), radius * sin(theta));
polar = fract(polar);

gl_FragColor = texture(tex, polar);
```
